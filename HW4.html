def point_to_point(start,size,t,bkgd):
    first_size_data = bkgd[start:start+size]
    plt.scatter(t[start:start+size],first_size_data)
    plt.title("Scatter Plot")
    plt.xlabel("time")
    plt.ylabel("background")
point_to_point(500,2000,t,bkgd)
