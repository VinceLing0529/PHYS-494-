sigsleft = np.zeros(40)
for i in range(1,41):
    maskH = ((90+i)<dictH['mass'])&(dictH['mass']<150)
    CUTH = dictH['mass'][maskH]
    maskQ = ((90+i)<dictQ['mass'])&(dictQ['mass']<150)
    CUTQ = dictQ['mass'][maskQ]
    nhiggs = (len(CUTH)/100000) * n_higgs
    nqcd = (len(CUTQ)/100000) * n_qcd
    sig_perc = stats.poisson.cdf(nqcd+nhiggs,mu=nqcd)
    sigsleft[i-1] = stats.norm.ppf(sig_perc)
